[•.wav/.flac to .mp3 converter]


@echo off
mkdir mp3

for %%f in (*.wav *.flac) do (
    echo Converting %%f...
    ffmpeg -y -i "%%f" -map_metadata 0 -vn -ab 320k "mp3\%%~nf.mp3"
)

echo Done!
pause
    

powershell -NoProfile -ExecutionPolicy Bypass -Command "Write-Host 'RedlonMatrix PowerShell'"
